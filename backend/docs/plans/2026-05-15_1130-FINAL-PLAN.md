## 🤖 Agent Pipeline Detail

### Pipeline Flow (agent/src/pipeline.ts)

```typescript
export async function runPipeline(
  jobId: string,
  photoUrls: string[],
  callbacks: {
    onProgress: (agent: string, step: string, message: string, toolCall?: string) => void
  }
): Promise<PipelineResult> {

  // Agent 1: Vision Analyzer
  callbacks.onProgress("vision", "start", "Memulai analisis foto...")
  const features = await runVisionAgent(photoUrls, callbacks)
  callbacks.onProgress("vision", "complete",
    `Face shape: ${features.face_shape} (${Math.round(features.face_confidence * 100)}%)`)

  // Agent 2: Style Matcher
  callbacks.onProgress("knowledge", "start", "Mencari gaya rambut yg cocok...")
  const candidates = await runKnowledgeAgent(features, callbacks)
  callbacks.onProgress("knowledge", "complete",
    `Ditemukan ${candidates.length} kandidat gaya rambut`)

  // Agent 3: Ranker & Explainer
  callbacks.onProgress("ranking", "start", "Menghitung skor & ranking...")
  const recommendations = await runRankerAgent(features, candidates, callbacks)
  callbacks.onProgress("ranking", "complete", "Top 3 rekomendasi siap!")

  // Agent 4: Image Generator (optional, if time permits)
  callbacks.onProgress("imagegen", "start", "Generate gambar referensi...")
  const images = await runImageGenAgent(recommendations, callbacks)
  // Merge images into recommendations
  for (const rec of recommendations) {
    const img = images.find(i => i.styleName === rec.style_name)
    if (img) rec.reference_image_url = img.url
  }
  callbacks.onProgress("imagegen", "complete", "Gambar referensi siap!")

  return { features, recommendations }
}
```

### Agent Execution Engine (agent/src/runner.ts)

```typescript
// Core pattern: autonomous loop with tool calling
export async function runAgent(config, userMessage, onStep) {
  const messages = [
    { role: "system", content: config.systemPrompt },
    { role: "user", content: userMessage }
  ]

  let iterations = 0
  while (iterations < config.maxIterations) {
    iterations++
    onStep({ type: "thinking", message: `${config.name} berpikir... (iterasi ${iterations})` })

    const response = await llm.chat(messages, config.tools)

    if (response.tool_calls) {
      for (const toolCall of response.tool_calls) {
        onStep({ type: "tool_call", message: `Memanggil ${toolCall.name}`, toolName: toolCall.name })
        const result = await tool.execute(toolCall.params)
        onStep({ type: "tool_result", message: `${toolCall.name} selesai` })
        messages.push({ role: "tool", content: JSON.stringify(result) })
      }
      continue  // Let agent process tool results
    }

    // No tool calls = final answer
    return parseOutput(response.content)
  }
}
```
