## 🔄 Full User Flow

### Step 1: Auth
```
User → POST /api/auth/register { username, password }
     → Response: { token, user }

User → POST /api/auth/login { username, password }
     → Response: { token, user }
```

### Step 2: Upload & Analyze
```
User → POST /api/analyze (Bearer token + multipart: 1-4 foto)
     → Backend:
        1. Validate auth
        2. Rate limit check (in-memory)
        3. Upload foto → R2 (convert to webp)
        4. Create job in DB (status: pending)
        5. Start agent pipeline (async, don't block)
        6. Return { jobId, status: "processing" }
```

### Step 3: Poll Progress
```
User → GET /api/status/:jobId (Bearer token)
     → Response: {
         jobId,
         status: "processing" | "completed" | "failed",
         currentStep: "vision" | "knowledge" | "ranking" | "done",
         progress: [
           { agent: "vision", step: "analyzing_front", message: "Menganalisis foto depan...", timestamp },
           { agent: "vision", step: "tool_call", message: "Memanggil analyze_image tool...", timestamp },
           { agent: "vision", step: "complete", message: "Face shape: oval (87%)", timestamp },
           ...
         ]
       }
```

### Step 4: Get Results
```
User → GET /api/result/:jobId (Bearer token)
     → Response: {
         jobId,
         features: {
           face_shape: "oval",
           hair_thickness: "thick",
           hair_texture: "straight",
           ...
         },
         recommendations: [
           {
             rank: 1,
```

## 🤖 Agent Pipeline Detail

### Pipeline Orchestration (src/agents/pipeline.ts)

```typescript
async function runPipeline(jobId: string, photoUrls: string[]) {
  // Step 1: Vision Agent
  updateJob(jobId, "vision")
  logAgent(jobId, "vision", "start", "Memulai analisis foto...")
  
  const features = await runVisionAgent(photoUrls, (step) => {
    logAgent(jobId, "vision", step.type, step.message, step.toolName)
  })
  
  logAgent(jobId, "vision", "complete", `Face shape: ${features.face_shape} (${features.face_confidence * 100}%)`)
  
  // Step 2: Knowledge Gatherer
  updateJob(jobId, "knowledge")
  logAgent(jobId, "knowledge", "start", "Mencari gaya rambut yg cocok...")
  
  const candidates = await runKnowledgeAgent(features, (step) => {
    logAgent(jobId, "knowledge", step.type, step.message, step.toolName)
  })
  
  logAgent(jobId, "knowledge", "complete", `Ditemukan ${candidates.length} kandidat`)
  
  // Step 3: Ranker
  updateJob(jobId, "ranking")
  logAgent(jobId, "ranking", "start", "Menghitung skor & ranking...")
  
  const recommendations = await runRankerAgent(features, candidates, (step) => {
    logAgent(jobId, "ranking", step.type, step.message, step.toolName)
  })
  
  logAgent(jobId, "ranking", "complete", "Top 3 rekomendasi siap!")
  
  // Save result
  saveResult(jobId, features, recommendations)
  updateJob(jobId, "completed")
}
```

### Agent Execution Pattern (src/agents/runner.ts)

```typescript
// Each agent = autonomous loop with tool calling
async function runAgent(config, userMessage, onStep) {
  const messages = [
    { role: "system", content: config.systemPrompt },
    { role: "user", content: userMessage }
  ]

  while (iterations < maxIterations) {
    // Call LLM with tools
    const response = await llm.chat(messages, config.tools)
    
    if (response.tool_calls) {
      // Execute each tool call
      for (const toolCall of response.tool_calls) {
        onStep({ type: "tool_call", message: `Calling ${toolCall.name}...` })
        const result = await tool.execute(toolCall.params)
        onStep({ type: "tool_result", message: `Done: ${toolCall.name}` })
        messages.push({ role: "tool", content: result })
      }
      continue  // Let agent process tool results
    }
    
    // No tool calls = final answer
    return parseOutput(response.content)
  }
}
```

## 🗄️ Database Schema (Neon PostgreSQL)

```sql
-- Users
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  username VARCHAR(50) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Jobs
CREATE TABLE jobs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id),
  status VARCHAR(20) DEFAULT 'pending',
    -- pending, processing, completed, failed
  current_step VARCHAR(50),
    -- vision, knowledge, ranking, done
  photos JSONB,           -- array of R2 URLs
  created_at TIMESTAMPTZ DEFAULT NOW(),
  completed_at TIMESTAMPTZ
);

-- Agent progress logs (KUNCI BUAT DEMO)
CREATE TABLE agent_logs (
  id SERIAL PRIMARY KEY,
  job_id UUID REFERENCES jobs(id),
  agent_name VARCHAR(50),     -- vision, knowledge, ranker
  step VARCHAR(100),          -- analyzing, tool_call, complete
  message TEXT,               -- human readable
  tool_call VARCHAR(100),     -- which tool
  tool_input JSONB,
  tool_output JSONB,
  reasoning TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Results
CREATE TABLE results (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id UUID REFERENCES jobs(id) UNIQUE,
  features JSONB,             -- FaceFeatures
  recommendations JSONB,      -- Top 3
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```
